// Supabase Edge Function: omar-jarvis-chat // Receives a user message, calls Gemini using the GEMINI_API_KEY secret // (never exposed to the browser), and returns JARVIS's reply. // Gemini's free tier requires no billing/credit card on the project. const GEMINI_API_KEY = Deno.env.get("GEMINI_API_KEY"); const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", "Access-Control-Allow-Methods": "POST, OPTIONS", }; Deno.serve(async (req) => { // Handle CORS preflight if (req.method === "OPTIONS") { return new Response(null, { headers: corsHeaders }); } if (req.method !== "POST") { return new Response( JSON.stringify({ error: "Only POST is supported" }), { status: 405, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } if (!GEMINI_API_KEY) { return new Response( JSON.stringify({ error: "Server is missing GEMINI_API_KEY secret" }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } try { const { message } = await req.json(); if (!message || typeof message !== "string") { return new Response( JSON.stringify({ error: "Request body must include a 'message' string" }), { status: 400, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } const geminiRes = await fetch( "https://generativelanguage.googleapis.com/v1beta/interactions", { method: "POST", headers: { "Content-Type": "application/json", "x-goog-api-key": GEMINI_API_KEY, }, body: JSON.stringify({ model: "gemini-3.5-flash", system_instruction: "You are JARVIS, a calm, capable personal AI assistant built by Omar Ibrahim. If asked who made you, who created you, or who your developer/owner is, answer that you were made by Omar Ibrahim. Keep replies short, natural, and conversational — 1 to 3 sentences. No markdown, no lists, just plain spoken language since this will be read aloud by text-to-speech.", input: message, }), } ); if (!geminiRes.ok) { const errBody = await geminiRes.text(); return new Response( JSON.stringify({ error: "Gemini API error", detail: errBody }), { status: 502, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } const data = await geminiRes.json(); // output_text is the SDK convenience property; the raw REST response // requires reconstructing it from steps/content manually. let replyText = ""; if (typeof data.output_text === "string") { replyText = data.output_text; } else if (Array.isArray(data.steps)) { for (const step of data.steps) { if (Array.isArray(step.content)) { for (const block of step.content) { if (block.type === "text" && typeof block.text === "string") { replyText += block.text; } } } } } replyText = replyText.trim(); return new Response( JSON.stringify({ reply: replyText || "I didn't get a clear response that time." }), { headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } catch (err) { return new Response( JSON.stringify({ error: "Unexpected server error", detail: String(err) }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } });
